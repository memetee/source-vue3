```html
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <title>Document</title>
  </head>
  <body>

      //   effect,
      //   computed,
      // } from '../dist/reactivity.esm.js'

      import {
        reactive,
        effect,
        ref,
        computed,
      } from '../../../node_modules/vue/dist/vue.esm-browser.prod.js'

      const count = ref(0)

      /**
       * computed 它比较特殊
       * 它既是一个sub，又是一个dep
       */
      const c = computed(() => {
        return count.value + 1
      })

      let e = effect(() => {
        console.log(c.value)
      })

      setTimeout(() => {
        count.value = 1
      }, 1000)
    </script>
  </body>
</html>
```

上述用了官方的 computed，当 count 发生改变的时候，computed 中的 count 也会发生变化并且重新执行，然后 c 发生更新，触发 effect 重新执行

## computed 的特性

computed 的特性是：

- computed 中回调的依赖发生变化，那么会重新执行，从这个角度来看，他是 sub
- c.value 发生了变化的时候，它又会触发 effect，从这个角度来看，它又是一个 dep

如果要画一个图的话，他是这样：[[8-计算属性-6ECE81F709640946DF1E44BC1FA1521F.excalidraw]]

从这个图可以很明显的看出来，count 这个 ref 发生了修改，会通知所有的依赖，在这里 count 的依赖就是 c，也就是 computed，computed 此时会更新，然后 return count + 1，之后 c 发生了变化，c 发生了变化，作为 dep，他也需要通知他的 sub，所以会通知 effect 重新执行了

```ts
import { isFunction } from '@vue/shared'
import { ReactiveFlags } from './ref'
import { Dependency, Link, Sub } from './system'

class ComputedRefImpl implements Dependency, Sub {
  // computed也是一个ref，通过isRef也返回true
  [ReactiveFlags.IS_REF] = true

  // 保存fn的返回值用
  _value

  /**
   * 作为 dep，要关联subs，等值更新了，通知更新
   * 订阅者链表的头节点
   */
  subs: Link

  // 订阅者链表的尾节点
  subsTail: Link

  /**
   * 作为 sub，要保存哪些 dep 被收集了
   * 依赖链表的头节点
   */
  deps: Link | undefined

  // 依赖链表的尾节点
  depsTail: Link | undefined

  // 是否被追踪
  tracking = false

  constructor(
    public fn, // getter
    private setter, // setter
  ) {}
}

/**
 * 计算属性
 * @param getterOrOptions 函数或者配置项，配置项的话有getter和setter
 */
export function computed(getterOrOptions) {
  let getter
  let setter
  if (isFunction(getterOrOptions)) {
    /**
     * const c = computed(() => {})
     */
    getter = getterOrOptions
  } else {
    /**
     * const c = computed({
     *  get() {},
     *  set() {}
     * })
     */
    getter = getterOrOptions.get
    setter = getterOrOptions.set
  }

  return new ComputedRefImpl(getter, setter)
}
```

## 实现 computed 的 sub

packages\reactivity\src\computed.ts

```ts
class ComputedRefImpl implements Dependency, Sub {
	...

  get value() {
    this.update()
    // 要和 sub 做关联关系
    if (activeSub) {
      link(this, activeSub)
    }
    return this._value
  }
	...

  update() {
    /**
     * 实现 sub 的功能，为了在执行 fn 期间，收集 fn 执行过程中访问到的响应式数据
     * 建立 dep 和 sub 之间的关联关系
     */

    // 先将当前的effect保存起来，用来处理嵌套的逻辑
    const prevSub = activeSub
    // 每次执行之前把this放到activeSub上，这样在依赖收集的时候就能知道当前正在执行的effect函数了
    setActiveSub(this) // 这里保存的是对象了

    // 开始追踪依赖，将depsTail尾结点设置成undefined
    startTrack(this)
    try {
      this._value = this.fn() // 直接执行了fn，这里拿到fn的返回值保存起来
    } finally {
      endTrack(this)

      // 执行完成之后把activeSub置空，这样在依赖收集的时候就知道没有正在执行的effect函数了
      setActiveSub(prevSub) // 这里恢复之前的effect函数，这样在处理嵌套的逻辑的时候就能正确地收集依赖了
    }
  }
}
```

setActiveSub 这个是设置 activeSub 的，重新配置

## 更新依赖报错

现在已经可以实现了 sub 和 dep 都建立了关联关系了，但是如果你现在把你的代码跑起来的话，会发现是报错的。
![[8-计算属性-CE32B138F6031830CADC4FEAE2CAF187.png]]
原因也很简单，因为在更新的时候是通过 notify，但是 computed 实没有 notify 的，所以这里报错了，那么要在哪里解决呢？

packages\reactivity\src\system.ts

```ts
...

function processComputedUpdate(sub) {
  /**
   * 更新计算属性
   * 1.调用 update
   * 2. 通知 subs链表上的所有的 sub，重新执行
   */
  sub.update()
  propagate(sub.subs)
}

export function propagate(subs) {
  let link = subs
  const queueEffect = []
  while (link) {
    const sub = link.sub

    if (!sub.tracking) {

	    // 如果sub中存在update，那么表示是computed，那么按照这种方式更新
      if ('update' in sub) {
        processComputedUpdate(sub)
      } else {
        queueEffect.push(sub)
      }
    }
    link = link.nextSub
  }
  queueEffect.forEach(effect => effect.notify())
}
```

这样的话是能解决上面的报错，采用的方案是如果是 computed，则单独用 processComputedUpdate 进行更新。

packages\reactivity\src\computed.ts

```ts

class ComputedRefImpl implements Dependency, Sub {

    ...

  /**
   * 计算属性脏不脏，如果dirty为 true， 表示计算属性是脏的，get value 的时候，需要执行update
   */
  dirty = true

  constructor(
    public fn, // getter
    private setter,
  ) {}

  get value() {
    if (this.dirty) {
      // 如果计算属性脏了，那么执行 update
      this.update()
    }

    /**
     * 要和 sub 做关联关系
     */
    if (activeSub) {
      link(this, activeSub)
    }
    return this._value
  }

	...


  update() {
        ...

        try {
		  this._value = this.fn()

          //已经更新完了，变成false
          this.dirty = false
        } finally {

            ...

        }
      }
    }
}
```

上面增加了一个 dirty 来做判断，他是一个标识符

```ts
function processComputedUpdate(sub) {
  // 如果subs有的时候在执行
  if (sub.subs && sub.update()) {
    /**
     * 更新计算属性
     * 1.调用 update
     * 2. 通知 subs链表上的所有的 sub，重新执行

    // sub.update返回true，表示值发生了变化
```

```ts
export function propagate(subs) {
  let link = subs
  const queueEffect = []
  while (link) {
    const sub = link.sub

    // 如果当前的订阅者没有被追踪，就跳过
    // 因为如果当前的订阅者没有被追踪，说明当前的订阅者没有被使用，所以不需要触发更新
    // 如果当前的订阅者被追踪，说明当前的订阅者被使用了，所以需要触发更新
    if (!sub.tracking && !sub.dirty) {
```

在 propagate 函数中，收集执行函数的过程中会判断，如果当前的 sub 已经标识为脏的了，表示它已经收集过了，那么不再收集触发更新了
紧接着将收集完的进行更新，当然，这个时候，肯定不会出现重复更新的了
更新完了以后，就把标识置为不脏了
